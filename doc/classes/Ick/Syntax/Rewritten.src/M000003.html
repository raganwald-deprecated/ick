      def generate_ruby sexp
        ruby = Ruby2Ruby.new.process(sexp)
        ruby.sub!(/\A(def \S+)\(([^\)]*)\)/, '\1 |\2|')     # move args
        ruby.sub!(/\Adef[^\n\|]+/, 'proc { ')               # strip def name
        ruby.sub!(/end\Z/, '}')                             # strip end
        ruby.gsub!(/\s+$/, '')                              # trailing WS bugs me
        ruby
      end
